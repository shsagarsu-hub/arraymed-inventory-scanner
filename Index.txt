  // ========== API CONFIG ==========
  // The Apps Script project is deployed as a pure JSON API (doGet/doPost).
  // Paste your deployment's /exec URL once — it's remembered in this browser.
  let API_BASE_URL = (function() {
    try { return localStorage.getItem('arraymed_inv_api_url') || ''; } catch (e) { return ''; }
  })();

  function setApiUrl(url) {
    API_BASE_URL = url.trim();
    try { localStorage.setItem('arraymed_inv_api_url', API_BASE_URL); } catch (e) {}
  }

  function apiGet(action, params) {
    if (!API_BASE_URL) return Promise.reject('API URL not configured — tap ⚙️ to set it.');
    const qs = Object.keys(params || {}).map(k => encodeURIComponent(k) + '=' + encodeURIComponent(params[k])).join('&');
    const url = API_BASE_URL + '?action=' + encodeURIComponent(action) + (qs ? '&' + qs : '');
    return fetch(url).then(r => {
      if (!r.ok) throw new Error('HTTP ' + r.status);
      return r.json();
    });
  }

  function apiPost(action, body) {
    if (!API_BASE_URL) return Promise.reject('API URL not configured — tap ⚙️ to set it.');
    return fetch(API_BASE_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'text/plain;charset=utf-8' }, // avoids CORS preflight on Apps Script
      body: JSON.stringify(Object.assign({ action: action }, body))
    }).then(r => {
      if (!r.ok) throw new Error('HTTP ' + r.status);
      return r.json();
    });
  }

  // ========== GS1 PARSER ==========
  const GS1_AIS = {
    '01': { label: 'GTIN', len: 14 },
    '02': { label: 'Content GTIN', len: 14 },
    '10': { label: 'Lot/Batch', len: null },
    '11': { label: 'Production Date', len: 6 },
    '13': { label: 'Pack Date', len: 6 },
    '15': { label: 'Best Before', len: 6 },

  function parseGS1(raw) {
    let data = raw.replace(/^\]d2/, '').replace(/^\]C1/, '').replace(/^\]e0/, '');
    const parenRegex = /\((\d{2,4})\)([^()]*)/g;
    let parenMatch;
    const fields = [];
    let usedParen = false;
    while ((parenMatch = parenRegex.exec(data)) !== null) {
      usedParen = true;
      const ai = parenMatch[1];
      const value = parenMatch[2];
      const info = GS1_AIS[ai];
      fields.push({ ai, label: info ? info.label : 'AI ' + ai, value: value.trim() });
    }
    if (usedParen && fields.length > 0) return { isGS1: true, fields };

    const parts = data.split(/\x1d/);
    for (const part of parts) {
      let pos = 0;
      while (pos < part.length) {
        let matched = false;
        for (const aiLen of [3, 2]) {
          if (pos + aiLen > part.length) continue;
          const ai = part.substring(pos, pos + aiLen);
          if (GS1_AIS[ai]) {
            const info = GS1_AIS[ai];
            pos += aiLen;
            let value;
            if (info.len) { value = part.substring(pos, pos + info.len); pos += info.len; }
            else { value = part.substring(pos); pos = part.length; }
            fields.push({ ai, label: info.label, value: value.trim() });
            matched = true;
            break;
          }
        }
        if (!matched) break;
      }
    }
    if (fields.length > 0) return { isGS1: true, fields };
    return { isGS1: false, fields: [{ ai: '', label: 'Barcode', value: data }] };
  }

  // Grabs a frame from the live <video> every ~350ms, decodes it with ZXing-WASM,
  // and on a hit processes the result. Mirrors the Logistics CRM's manual
  // canvas-capture approach instead of relying on a library's built-in camera loop.
  async function runScanLoop() {
    const video = document.getElementById('camVideo');
    const canvas = document.getElementById('camCanvas');
    const ctx = canvas.getContext('2d', { willReadFrequently: true });

    async function tick() {
      if (!isScanning) return;
      if (video.readyState === video.HAVE_ENOUGH_DATA && video.videoWidth > 0) {
        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
        try {
          const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
          const results = await ZXingWASM.readBarcodesFromImageData(imageData, {
            tryHarder: true,
            formats: ['DataMatrix', 'Code128', 'Code39', 'EAN-13', 'EAN-8', 'QRCode', 'ITF', 'Codabar', 'UPC-A', 'UPC-E', 'PDF417', 'Aztec'],
            maxSymbols: 1
          });
          if (results && results.length > 0 && results[0].text) {
            handleDecodedBarcode(results[0].text, results[0].format);
          }
        } catch (e) {
          // decode miss on this frame — normal, just keep looping
        }
      }
      if (isScanning) scanLoopHandle = requestAnimationFrame(tick);
    }
    scanLoopHandle = requestAnimationFrame(tick);
  }

  function handleDecodedBarcode(decodedText, formatName) {
    const now = Date.now();
    if (now - lastScanTime < 2000 && decodedText === lastScanCode) return;
    lastScanTime = now;
    lastScanCode = decodedText;
    beep();

    const parsed = parseGS1(decodedText);
    const gtinField = parsed.fields.find(f => f.ai === '01');
    const gtin = gtinField ? gtinField.value : decodedText;
    const lotField = parsed.fields.find(f => f.ai === '10');
    const expiryField = parsed.fields.find(f => f.ai === '17');
    const prodField = parsed.fields.find(f => f.ai === '11');

    currentParsed = {
      gtin, raw: decodedText, format: formatName || 'Unknown', isGS1: parsed.isGS1,
      lot: lotField ? lotField.value : '',
      expiry: expiryField ? formatGS1Date(expiryField.value) : '',
      productionDate: prodField ? formatGS1Date(prodField.value) : '',
      fields: parsed.fields
    };

    renderParsedFields(parsed, formatName || 'Unknown');
    document.getElementById('scanResultCard').style.display = 'block';
    document.getElementById('scanResultCard').scrollIntoView({ behavior: 'smooth', block: 'start' });

    currentQty = 1;
    document.getElementById('qtyVal').textContent = '1';
    document.getElementById('itemMsgOk').style.display = 'none';
    document.getElementById('itemMsgErr').style.display = 'none';

    lookupItem(gtin);
  }

  function renderParsedFields(parsed, formatName) {
    let html = '<div class="parsed-grid">';
    for (const f of parsed.fields) {
      const displayValue = isDateField(f.ai) ? formatGS1Date(f.value) : f.value;
      const isGtin = f.ai === '01';
      const fullWidth = f.ai === '01' || f.label === 'Barcode';
      html += '<div class="parsed-field ' + (fullWidth ? 'full' : '') + '"><div class="parsed-label">' + f.label + (f.ai ? ' (' + f.ai + ')' : '') + '</div><div class="parsed-value ' + (isGtin ? 'highlight' : '') + '">' + displayValue + '</div></div>';
    }
    html += '<div class="parsed-field"><div class="parsed-label">Format</div><div class="parsed-value">' + (parsed.isGS1 ? 'GS1 ' : '') + formatName + '</div></div>';
    html += '</div>';
    document.getElementById('parsedFields').innerHTML = html;
  }

  function lookupItem(gtin) {
    const badge = document.getElementById('itemStatusBadge');
    badge.innerHTML = '<div class="item-status unknown">🔍 Checking item master…</div>';
    apiGet('lookupItemMaster', { gtin: gtin })
      .then(res => {
        if (res && res.found) {
          badge.innerHTML = '<div class="item-status known">✅ Known item — auto-filled from Item Master</div>';
          document.getElementById('itemNameInput').value = res.itemName || '';
          document.getElementById('brandInput').value = res.brand || '';
        } else {
          badge.innerHTML = '<div class="item-status unknown">⚠️ New GTIN — please enter the Tally Item Name once, it will be remembered next time</div>';
          document.getElementById('itemNameInput').value = '';
          document.getElementById('brandInput').value = '';
        }
      })
      .catch(e => {
        badge.innerHTML = '<div class="item-status unknown">⚠️ Could not check item master — enter name manually</div>';
      });
  }

  function submitScan() {
    const itemName = document.getElementById('itemNameInput').value.trim();
    const brand = document.getElementById('brandInput').value.trim();
    const remarks = document.getElementById('remarksInput').value.trim();
    const scannedBy = document.getElementById('scannedByInput').value.trim();
    const errEl = document.getElementById('itemMsgErr');
    const okEl = document.getElementById('itemMsgOk');

    if (!itemName) {
      errEl.textContent = 'Please enter the Tally Item Name before logging.';
      errEl.style.display = 'block'; okEl.style.display = 'none';
      return;
    }
    if (!currentParsed) return;

    const btn = document.getElementById('logScanBtn');
    btn.disabled = true; btn.textContent = '…Saving';

    // Save / update item master mapping first, then log the scan
    apiPost('saveItemMaster', { gtin: currentParsed.gtin, itemName: itemName, brand: brand, notes: '' })
      .then(() => {
        return apiPost('logScan', {
          entry: {
            direction: currentDirection,
            gtin: currentParsed.gtin,
            itemName: itemName,
            lot: currentParsed.lot,
            expiry: currentParsed.expiry,
            productionDate: currentParsed.productionDate,
            qty: currentQty,
            format: currentParsed.format,
            raw: currentParsed.raw,
            scannedBy: scannedBy,
            remarks: remarks
          }
        });
      })
      .then(res => {
        btn.disabled = false; btn.textContent = '✓ Log This Scan';
        if (res && res.success) {
          okEl.textContent = '✅ ' + currentDirection + ' scan logged for "' + itemName + '" (Qty: ' + currentQty + ')';
          okEl.style.display = 'block'; errEl.style.display = 'none';
          setTimeout(() => { okEl.style.display = 'none'; }, 3000);
        } else {
          errEl.textContent = '❌ ' + ((res && res.message) || 'Failed to log scan.');
          errEl.style.display = 'block';
        }
      })
      .catch(e => {
        btn.disabled = false; btn.textContent = '✓ Log This Scan';
        errEl.textContent = '❌ Connection error: ' + e;
        errEl.style.display = 'block';
      });
  }

  function fmtNum(n) {
    n = parseFloat(n) || 0;
    return n.toLocaleString('en-IN');
  }

  function loadDashboard() {
    apiGet('getDashboardData', {})
      .then(res => {
        if (!res || !res.success) return;
        const k = res.kpis;
        document.getElementById('kTotalScans').textContent = fmtNum(k.totalScans);
        document.getElementById('kUniqueItems').textContent = fmtNum(k.uniqueItems);
        document.getElementById('kTotalIn').textContent = fmtNum(k.totalIn);
        document.getElementById('kTotalOut').textContent = fmtNum(k.totalOut);
        document.getElementById('kNegative').textContent = fmtNum(k.negativeCount);
        document.getElementById('kNetBal').textContent = fmtNum(k.totalIn - k.totalOut);

        const stockBody = document.getElementById('stockTableBody');
        if (!res.stockRows.length) {
          stockBody.innerHTML = '<tr><td colspan="7"><div class="empty-state">No scans yet. Start scanning to see live stock here.</div></td></tr>';
        } else {
          stockBody.innerHTML = res.stockRows.map(r => {
            const neg = r.balance < 0;
            return '<tr class="' + (neg ? 'neg' : '') + '">' +
              '<td style="font-weight:700;">' + esc(r.itemName) + '</td>' +
              '<td>' + esc(r.lot || '—') + '</td>' +
              '<td>' + esc(r.expiry || '—') + '</td>' +
              '<td style="color:var(--green);font-weight:700;">' + fmtNum(r.qtyIn) + '</td>' +
              '<td style="color:var(--red);font-weight:700;">' + fmtNum(r.qtyOut) + '</td>' +
              '<td class="bal" style="font-weight:800;">' + fmtNum(r.balance) + '</td>' +
              '<td style="font-size:10px;color:var(--text3);">' + fmtDate(r.lastScanned) + '</td>' +
            '</tr>';
          }).join('');
        }

        const logsList = document.getElementById('recentLogsList');
        if (!res.recentLogs.length) {
          logsList.innerHTML = '<div class="empty-state">No activity yet.</div>';
        } else {
          logsList.innerHTML = res.recentLogs.map(l => {
            return '<div class="log-entry">' +
              '<div class="log-top">' +
                '<span class="log-name">' + esc(l.itemName || '(Unmapped GTIN: ' + l.gtin + ')') + '</span>' +
                '<span class="log-dir ' + (l.direction === 'OUT' ? 'out' : 'in') + '">' + esc(l.direction) + '</span>' +
              '</div>' +
              '<div class="log-meta">Qty: ' + fmtNum(l.qty) + (l.lot ? ' · Lot: ' + esc(l.lot) : '') + (l.expiry ? ' · Exp: ' + esc(l.expiry) : '') + (l.scannedBy ? ' · By: ' + esc(l.scannedBy) : '') + '</div>' +
              '<div class="log-time">' + fmtDate(l.timestamp) + '</div>' +
            '</div>';
          }).join('');
        }
      });
  }

  function fmtDate(d) {
    if (!d) return '—';
    try {
      const dt = new Date(d);
      return dt.toLocaleDateString('en-IN') + ' ' + dt.toLocaleTimeString('en-IN', { hour: '2-digit', minute: '2-digit' });
    } catch (e) { return String(d); }
  }

  function esc(s) {
    return (s || '').toString().replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  }
